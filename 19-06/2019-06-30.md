# Linux扩容/分区

```
df -lh
fdisk -l
fdisk /dev/sda 
reboot
mkfs.ext4 /dev/sda3
pvcreate /dev/sda3
vgdisplay
gvextend zu /dev/sda3
lvdisplay
lvextend -l +100%FREE /dev/vg_ysqcentos/lv_root
e2fsck -f /dev/vg_ysqcentos/lv_root
resize2fs /dev/vg_ysqcentos/lv_root
df -lh
```
